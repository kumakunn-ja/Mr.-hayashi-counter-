<script>
    // カウントの初期値
    let chalkData = {
        teacher: "林田先生",
        crashCount: 0,
        lastUpdated: ""
    };

    const countDisplay = document.getElementById('countDisplay');

    // 画面の数値を更新する関数
    function updateDisplay() {
        countDisplay.textContent = chalkData.crashCount;
    }

    // カウントアップ
    function countUp() {
        chalkData.crashCount++;
        chalkData.lastUpdated = new Date().toLocaleString('ja-JP');
        updateDisplay();
    }

    // リセット
    function resetCount() {
        if(confirm("カウントを0に戻してもよろしいですか？")) {
            chalkData.crashCount = 0;
            chalkData.lastUpdated = new Date().toLocaleString('ja-JP');
            updateDisplay();
        }
    }

    // JSONファイルをダウンロード保存する
    function downloadJSON() {
        const jsonString = JSON.stringify(chalkData, null, 2);
        const blob = new Blob([jsonString], { type: "application/json" });
        const url = URL.createObjectURL(blob);
        
        const a = document.createElement('a');
        a.href = url;
        a.download = `hayashida_chalk_count.json`;
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        URL.revokeObjectURL(url);
    }

    // JSONファイルを読み込んでデータを復元する
    function loadJSON(input) {
        const file = input.files[0];
        if (!file) return;

        const reader = new FileReader();
        reader.onload = function(e) {
            try {
                const parsedData = JSON.parse(e.target.result);
                if (parsedData && typeof parsedData.crashCount === 'number') {
                    chalkData = parsedData;
                    updateDisplay();
                    alert(`データを読み込みました！現在のクラッシュ数: ${chalkData.crashCount}回`);
                } else {
                    alert("無効なJSONファイルです。");
                }
            } catch (err) {
                alert("ファイルの読み込みに失敗しました。");
            }
        };
        reader.readAsText(file);
        // 同じファイルを連続で読み込めるようにリセット
        input.value = '';
    }

    // 初期表示
    updateDisplay();
</script>
